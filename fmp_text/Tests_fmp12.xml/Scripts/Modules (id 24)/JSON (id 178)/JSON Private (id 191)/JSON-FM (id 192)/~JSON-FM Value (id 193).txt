################################################################################
# PURPOSE:
#    Parse a JSON value. It could be an object, an array, a string, a number, or a word.
#
# PARAMETERS: none
#
# RETURNS:
#    A parsed value.
#
# REVISIONS:
#    2013-AUG-06	 Daniel Smith dansmith65@gmail.com	- created
################################################################################
#
#
Perform Script [ “~JSON-FM White” ]
If [  ]
Else If [ $$~JSON.CH = "{" ]
Perform Script [ “~JSON-FM Object” ]
Else If [ $$~JSON.CH = "[" ]
Perform Script [ “~JSON-FM Array” ]
Else If [ $$~JSON.CH = "\"" ]
Perform Script [ “~JSON-FM String” ]
Else If [ $$~JSON.CH = "-" ]
Perform Script [ “~JSON-FM Number” ]
Else If [ $$~JSON.CH >= "0"and $$~JSON.CH <= "9" ]
Perform Script [ “~JSON-FM Number” ]
Else
Perform Script [ “~JSON-FM Word” ]
End If
#
Exit Script [ Result: Get ( ScriptResult ) ]
